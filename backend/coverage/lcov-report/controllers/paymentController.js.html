async function getTopRevenueGames(req, res) {
    try {
        const topRevenueGames = await game_details.aggregate([
            {
                $project: {
                    game_name: 1,

                    quantity_sold: 1,
                    totalRevenue: { $multiply: ["$price", "$quantity_sold"] }
